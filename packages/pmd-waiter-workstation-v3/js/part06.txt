ngify({code:code, amount:paymentBase()})});
      state.payment.coupon = payload;
      renderAdjustments();
      renderPaymentTotals();
      toast('Coupon applied');
    } catch (error) {
      if (result) { result.textContent = error.message || 'Coupon invalid'; result.style.color = '#ff6578'; }
      toast(error.message || 'Coupon invalid.', true);
    }
  }

  function guestCheckoutUrl() {
    var summary = state.payment.summary;
    if (!summary || !summary.guest_checkout_url) return null;
    var url = new URL(summary.guest_checkout_url, location.origin);
    if (state.payment.method.indexOf('online:') === 0) url.searchParams.set('payment_method', state.payment.method.split(':').slice(1).join(':'));
    url.searchParams.set('source', 'waiter_workstation_v3');
    return url.toString();
  }

  async function copyPaymentLink() {
    var url = guestCheckoutUrl();
    if (!url) return toast('Customer payment link unavailable.', true);
    try { await navigator.clipboard.writeText(url); toast('Customer payment link copied'); }
    catch (error) { window.prompt('Copy customer payment link:', url); }
  }

  async function executePayment() {
    if (!state.payment.summary || state.payment.submitting) return;
    var order = activeOrder();
    var online = state.payment.method.indexOf('online:') === 0;
    if (online) {
      var checkout = guestCheckoutUrl();
      if (!checkout) return toast('Customer checkout unavailable.', true);
      window.open(checkout, '_blank', 'noopener');
      return toast('Secure checkout opened. Refresh status after payment.');
    }
    state.payment.submitting = true;
    renderPaymentTotals();
    try {
      if (state.payment.method === 'direct_terminal') {
        var terminalResult = await fetchJson(route(urls.terminalPayment, order.order_id), {method:'POST', headers:{'Content-Type':'application/json'}, body:JSON.stringify({provider_code:state.payment.provider})}, 30000);
        toast(terminalResult.message || 'Terminal request sent');
        await loadPaymentSummary(true);
        return;
      }
      var splitMode = state.payment.split === 'shares' ? 'custom' : state.payment.split;
      var result = await fetchJson(route(urls.paymentSettle, order.order_id), {
        method:'POST', headers:{'Content-Type':'application/json'}, body:JSON.stringify({
          idempotency_key:state.payment.idempotency,
          payment_method:state.payment.method,
          provider_code:state.payment.method === 'external_terminal' ? 'external_terminal' : null,
          split_mode:splitMode,
          amount:paymentBase(),
          selected_items:state.payment.split === 'items' ? paymentSelectedItems() : null,
          tip_amount:paymentTip(),
          coupon_code:state.payment.coupon ? state.payment.coupon.code : null,
          payer_label:state.payment.payer,
          payment_reference:state.payment.reference,
          cash_received:state.payment.method === 'cash' ? num(state.payment.tender,paymentPayable()) : null,
          external_confirmed:state.payment.method === 'external_terminal' ? state.payment.externalConfirmed : false,
          expected_remaining:state.payment.summary.settlement.remaining_amount,
          expected_updated_at:state.payment.summary.order.updated_at
        })
      }, 30000);
      state.payment.summary = result.summary || state.payment.summary;
      state.payment.idempotency = nowId('payment');
      state.payment.coupon = null;
      state.payment.couponCode = '';
      state.payment.tip = 0;
      state.payment.customTip = '';
      state.payment.reference = '';
      state.payment.externalConfirmed = false;
      state.payment.tender = roundMoney(paymentRemaining()).toFixed(2);
      state.metrics.payments += 1;
      renderPayment();
      await refreshTable(true);
      await loadLauncher(true);
      toast(result.message || 'Payment recorded');
    } catch (error) {
      toast(error.message || 'Payment failed.', true);
      if (error.status === 409 || error.status === 422) await loadPaymentSummary(true);
    } finally {
      state.payment.submitting = false;
      renderPaymentTotals();
    }
  }

  function keypad(key) {
    var value = String(state.payment.tender || '');
    var payable = paymentPayable();
    if (/^\d$/.test(key) && value === payable.toFixed(2)) value = '';
    if (key === 'clear') value = '';
    else if (key === 'back') value = value.slice(0, -1);
    else if (key === 'exact') value = payable.toFixed(2);
    else if (key === 'plus5') value = (Math.ceil(payable / 5) * 5).toFixed(2);
    else if (key === 'plus10') value = (Math.ceil(payable / 10) * 10).toFixed(2);
    else if (key === '00') value += '00';
    else if (/^\d$/.test(key)) value += key;
    if (value.length > 8) value = value.slice(0, 8);
    state.payment.tender = value;
    renderPaymentTotals();
  }

  /* Operations */
  async function loadOperations() {
    var order = activeOrder();
    if (!order) return toast('No active order yet.', true);
    showDrawer('operations');
    var body = one('[data-ws3-operations-body]');
    if (body) body.innerHTML = '<div class="pmd-ws3-empty">LOADING OPERATIONS…</div>';
    try {
      state.operations = await fetchJson(route(urls.operations, order.order_id) + '?_=' + Date.now(), {}, 18000);
      state.metrics.operationLoads += 1;
      renderOperations();
    } catch (error) {
      if (body) body.innerHTML = '<div class="pmd-ws3-empty">' + esc(error.message || 'Operations unavailable') + '</div>';
      toast(error.message || 'Operations unavailable.', true);
    }
  }

  }
