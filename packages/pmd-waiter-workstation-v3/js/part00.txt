      return {'&':'&amp;','<':'&lt;','>':'&gt;','"':'&quot;',"'":'&#039;'}[character];

    if (!timestamp) return '';

    if (minutes < 1) return 'JUST OPENED';

    var hours = Math.floor(minutes / 60);
    var remain = minutes % 60;
    return hours + 'H' + (remain ? ' ' + remain + 'M' : '');
  }

    return Math.abs(hash) % COLORS;

  async function fetchJson(url, options, timeoutMs) {
    var controller = new AbortController();
    var timer = setTimeout(function () { controller.abort(); }, timeoutMs || 15000);
    var config = Object.assign({
      credentials: 'same-origin',
      cache: 'no-store',
      signal: controller.signal,
      headers: {
        'Accept': 'application/json',
        'X-Requested-With': 'XMLHttpRequest',
        'X-CSRF-TOKEN': csrf()
      }
    }, options || {});
    if (options && options.headers) config.headers = Object.assign({}, config.headers, options.headers);
    try {
      var response = await fetch(url, config);
      var payload = await response.json().catch(function () { return {}; });
      if (!response.ok || payload.ok === false) {
        var error = new Error(payload.message || payload.error || ('HTTP ' + response.status));
        error.status = response.status;
        error.payload = payload;
        throw error;
      }
      return payload;
    } catch (error) {
      if (error && error.name === 'AbortError') throw new Error('The request took too long. Please retry.');
      throw error;
    } finally {
      clearTimeout(timer);
    }
  }
